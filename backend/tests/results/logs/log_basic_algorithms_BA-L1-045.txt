def fibonacci_sum(n):
    # Initialize the first two Fibonacci numbers
    a, b = 0, 1
    # Initialize the sum to zero
    sum = 0
    # Iterate over the first n Fibonacci numbers
    for _ in range(n):
        # Add the current Fibonacci number to the sum
        sum += a
        # Update the values of a and b
        a, b = b, a + b
    # Return the sum of the first n Fibonacci numbers
    return sum
