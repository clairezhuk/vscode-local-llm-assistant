def reverse_vowels(s):
    # Define a set of vowels
    vowels = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'}
    # Reverse the string while keeping the vowels in their original positions
    return ''.join([s[i] if s[i] not in vowels else s[-i-1] for i in range(len(s))])
