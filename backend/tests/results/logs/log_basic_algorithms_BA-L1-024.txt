def gcd(a, b):
    # Base case: greatest common divisor of 0 or 1 is 0
    if b == 0:
        return a
    # Recursive case: gcd(a, b) = gcd(b, a % b)
    else:
        return gcd(b, a % b)

def gcd_array(arr):
    # Initialize the first element as the gcd of the entire array
    result = arr[0]
    for num in arr[1:]:
        result = gcd(result, num)
    return result
