def fib(n):
    # Initialize a cache to store the results of previous calls
    cache = {0: 0, 1: 1}
    
    # Define a helper function to get the value of a given number
    def get_value(num):
        # Check if the value is already in the cache
        if num in cache:
            return cache[num]
        # Calculate the value using the Fibonacci recurrence relation
        else:
            value = get_value(num - 1) + get_value(num - 2)
            # Store the calculated value in the cache
            cache[num] = value
            return value
    
    # Call the helper function with the input number
    return get_value(n)

# Example usage
print(fib(10))  # Output: 55
