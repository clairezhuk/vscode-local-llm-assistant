def gcd(a, b):
    # Base case: greatest common divisor of 0 or 1 is 0
    if b == 0:
        return a
    # Recursive case: gcd(a, b) = gcd(b, a % b)
    else:
        return gcd(b, a % b)

def lcm(a, b):
    # Calculate LCM using the formula: lcm(a, b) = abs(a*b) // gcd(a, b)
    return abs(a*b) // gcd(a, b)

def lcm_array(arr):
    result = arr[0]
    for num in arr[1:]:
        result = lcm(result, num)
    return result
