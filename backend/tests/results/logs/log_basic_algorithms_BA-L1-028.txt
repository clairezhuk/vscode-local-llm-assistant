```python
def bubble_sort_strings(arr):
    # Use a bubble sort algorithm to sort the array of strings by length
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if len(arr[j]) > len(arr[j+1]):
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# Test the function
```
