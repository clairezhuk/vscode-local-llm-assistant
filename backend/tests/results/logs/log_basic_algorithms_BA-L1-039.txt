def reverse_array(arr):
    # Calculate the length of the array
    n = len(arr)
    # Use two pointers to swap elements from the start and the end of the array
    for i in range(n // 2):
        # Swap the elements at the current and opposite positions
        arr[i], arr[n - i - 1] = arr[n - i - 1], arr[i]
