def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    # Perform binary search to find the target in the array
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    # If the target is not found, return -1
    return -1
