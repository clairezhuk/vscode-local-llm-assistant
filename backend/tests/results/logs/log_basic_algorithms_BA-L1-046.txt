def gcd_multiple(arr):
    # Get the first number in the array
    gcd = arr[0]
    # Iterate over the rest of the numbers in the array
    for num in arr[1:]:
        # Calculate GCD with the current number
        gcd = gcd_multiple([gcd, num])
    # Return the GCD of the array
    return gcd
